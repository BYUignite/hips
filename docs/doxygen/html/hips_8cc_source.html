202std::vector<double> hips::projection(std::vector<double> &vcfd, std::vector<double> &weight) {
203
204 std::vector<double> xc = setGridCfd(weight);
205 std::vector<double> xh = setGridHips(nparcels);
206
207 int nc = xc.size() - 1;
208 int nh = xh.size() - 1;
209
210 std::vector<double> vh(nh, 0.0);
211 int jprev = 0;
212
213 for(int i = 0; i < nh; i++) {
214 for(int j = jprev + 1; j <= nc; ++j) {

216 double d1 = xc[j] - xc[j - 1];

225 double d2 = xh[i + 1] - xh[i];
226 double d = std::min(d1, d2);

252std::pair<std::vector<double>, std::vector<double>> hips::projection(std::vector<double> &vcfd, std::vector<double> &weight, const std::vector<double> &density) {
253
254 std::vector<double> xc = setGridCfd(weight);
255 std::vector<double> xh = setGridHips(nparcels);
256
257 // Initialize variables
258 int nc = xc.size() - 1;
259 int nh = xh.size() - 1;
260
261 std::vector<double> vh(nh, 0.0);
262 std::vector<double> rho_h(nh, 0.0);
263 int jprev = 0;
264
265 // Main loop
266 for (int i = 0; i < nh; i++) {
267 double total_dx = 0.0;
268
269 for (int j = jprev + 1; j <= nc; ++j) {
270 if (xc[j] <= xh[i + 1]) {
271 double d = std::min(xc[j] - xc[j - 1], xc[j] - xh[i]);
272 total_dx += d;
273 rho_h[i] += density[j - 1] * d;
274 vh[i] += density[j - 1] * vcfd[j - 1] * d;
275 }
276
277 else {
278 double d = std::min(xh[i + 1] - xc[j - 1], xh[i + 1] - xh[i]);
279 total_dx += d;
280 rho_h[i] += density[j - 1] * d;
281 vh[i] += density[j - 1] * vcfd[j - 1] * d;
282 jprev = j - 1;
283 break;
284 }
285 }
286
287 // Normalize results
288 rho_h[i] /= (xh[i + 1] - xh[i]);
289 vh[i] /= rho_h[i] * (xh[i + 1] - xh[i]);
290 }
291
292 return {vh, rho_h};
293}

354void hips::calculateSolution(const double tRun) {
355
356 unsigned long long nEddies = 0; // number of eddy events
357 int fileCounter = 0; // number of data files written
358 int iLevel; // tree level of EE with top at iLevel=0
359 int iTree; // one of two subtrees involved in swap at iLevel
360 double dtEE; // time increment to next eddy event
361 time = 0.0; // initialize simulation time
362
363 sample_hips_eddy(dtEE, iLevel); // get first EE at time 0+dtEE
364 nEddies++;
365
366 while (time+dtEE<=tRun) {
367 time += dtEE;
368 selectAndSwapTwoSubtrees(iLevel, iTree);
369 advanceHips(iLevel, iTree); // reaction and micromixing (if needed) to t=time
370
371 sample_hips_eddy(dtEE, iLevel);
372
373 nEddies++;
374 //cout<<"-------------------------------------"<<nEddies<<endl;
375 if(nEddies %2 == 0) writeData(++fileCounter, time);
376 }
377 time = tRun;
378 iLevel = 0; iTree = 0;
379 if(performReaction)
380 reactParcels_LevelTree(iLevel, iTree); // react all parcels up to end time
381}

390void hips::sample_hips_eddy(double &dtEE, int &iLevel) {
391
392 static double c1 = 1.0 - pow(2.0, 5.0/3.0*(iEta+1));
393 static double c2 = pow(2.0, Nm2) - pow(2.0, iEta+1);
394 static double c3 = pow(2.0, iEta+1);
395
396 //--------------- time to next eddy
397
398 double r = rand.getRand();
399 dtEE = -log(r)/eddyRate_total;
400
401 //----------------- get eddy level
402
403 r = rand.getRand();
404
405 if ( r <= eddyRate_inertial/eddyRate_total) { // inertial region
406 r = rand.getRand();
407 iLevel = ceil(3.0/5.0*log2(1.0-r*c1) - 1.0);
408 if (iLevel < 0) iLevel = 0;
409 if (iLevel > iEta) iLevel = iEta;
410 }
411
412 else { // "Batchelor" region
413 r = rand.getRand();
414 iLevel = ceil(log2(r*c2 + c3) - 1.0);
415 if(iLevel < iEta+1) iLevel = iEta+1;
416 if(iLevel > Nm3) iLevel = Nm3;
417 }
418
419 return;
420}

472void hips::selectAndSwapTwoSubtrees(const int iLevel, int &iTree) {
473
474 iTree = rand.getRandInt((1 << iLevel)-1);
475 int zero_q = rand.getRandInt(1); // 0q where q is 0 or 1
476 int one_r = 2 + rand.getRandInt(1); // 1r where r is 0 or 1
477
478 int Qstart = (zero_q << (Nm3-iLevel)) + (iTree << (Nm1-iLevel)); // starting index of Q parcels
479 int Rstart = (one_r << (Nm3-iLevel)) + (iTree << (Nm1-iLevel)); // starting index of R parcels
480 int nPswap = 1 << (Nm3-iLevel); // number of parcels that will be swapped
481
482 int Qend = Qstart + nPswap; // inclusive indices are Qstart to Qend-1
483 int Rend = Rstart + nPswap; // inclusive indices are Rstart to Rend-1
484 vector<int> aa(pLoc.begin()+Qstart, pLoc.begin()+Qend);
485 copy(pLoc.begin()+Rstart, pLoc.begin()+Rend, pLoc.begin()+Qstart); // python: pLoc[Qstart:Qend]=pLoc[Rstart:Rend]
486 copy(aa.begin(), aa.end(), pLoc.begin()+Rstart); // python: pLoc[Rstart:Rend]=aa
487}

632void hips::forceProfile() {
633 // Loop through each variable in the HiPS profile
634 for (int k = 0; k < varData.size(); k++) {
635 double s; // Temporary variable for summation
636
637 //---------- Force the left half of parcels to average 0 ----------
638
639 for (int i = 0; i < nparcels >> 1; i++)
640 s += varData[k][0][pLoc[i]]; // Calculate the sum of values in the left half of parcels
641
642 s /= (nparcels >> 1); // Calculate the average of values in the left half of parcels
643
644 for (int i = 0; i < nparcels >> 1; i++)
645 varData[k][0][pLoc[i]] += (-s - 0.0); // Adjust values in the left half of parcels to achieve an average of 0
646
647 //---------- Force the right half of parcels to average 1 ----------
648 s = 0.0;
649
650 for (int i = nparcels >> 1; i < nparcels; i++)
651 s += varData[k][0][pLoc[i]]; // Calculate the sum of values in the right half of parcels
652
653 s /= (nparcels >> 1); // Calculate the average of values in the right half of parcels
654
655 for (int i = nparcels >> 1; i < nparcels; i++)
656 varData[k][0][pLoc[i]] += (-s + 1.0); // Adjust values in the right half of parcels to achieve an average of 1
657 }
